    `name` VARCHAR(255) NOT NULL,
    `email` VARCHAR(255) NOT NULL,
    `user_profile` VARCHAR(255) DEFAULT NULL,

    FOREIGN KEY (`user_id`) REFERENCES `user_form`(`id`) ON DELETE CASCADE
